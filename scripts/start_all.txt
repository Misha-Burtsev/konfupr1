# Старт: покажи корень
ls
# cd без аргументов — в корень
cd
ls /
# Относительные и абсолютные пути
cd level1
ls
cd ..
cd /level1/level2
ls
cd ../..
# Папка с пробелами (если есть во VFS)
ls "folder with spaces"
cd "folder with spaces"
ls
cd ..
# Неизвестная команда
unknown
# Кривые кавычки (ошибка shlex)
bad "arg
# Переход в несуществующий каталог
cd NO_SUCH_DIR
# ls по несуществующему пути
ls nosuch
# ls по файлу (если есть readme.txt в корне — покажет (файл), иначе сообщение об ошибке)
ls readme.txt
# Попытка exit из скрипта (скрипт не падает, печатать должен SCRIPT ERROR)
exit
# Продолжаем
ls
